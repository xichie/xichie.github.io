网上已经有很多解析了，就不废话了。
主要注意一下几点：

 1. P、A、T三个字符都要出现。（测试点2）
 2. P、T仅且出现一次。
 3. P在T之前。
 4. P和T之间至少有一个A。
 5. P左边A的数量 * P和T之间A的数量 = T右边A的数量。
 
 
代码简单易懂，就是判断上面几个条件。

```c
#include <stdio.h>

int isRight(char *s)
{
	int i = 0,j = 0, count_p = 0, count_t = 0;
	int arr[3] = {0};
	while(s[i] != '\0')
	{
		switch(s[i])
		{
		case 'A':
			if(j > 2)
			{
				return 0;
			}
			arr[j] += 1;
			break;
		case 'P':
			count_p++;
			if(count_p > 1 || j >= 1)
			{
				return 0;
			}
			j++;
			break;
		case 'T':
			count_t++;
			if(count_t > 1 || j < 1)
			{
				return 0;
			}
			j++;
			break;
		default:
			return 0;
		}
		i++;
	}
	if(arr[1] != 0 && arr[0] * arr[1] == arr[2] && count_t && count_p)
	{
		return 1;
	}
	return 0;
}


int main()
{	
	int n, i;
	char str[100];
	scanf("%d", &n);
	for(i = 0; i < n; i++)
	{
		scanf("%s", str);
		if(isRight(str))
		{
			printf("YES\n");
		}
		else
		{
			printf("NO\n");
		}
	}
	return 0;
}```
