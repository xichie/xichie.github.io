第一次提交测试点2没通过。
试了半天，发现是当m > n的情况没有考虑到。
所以代码加了一行m = m % n。

```c
#include <stdio.h>
#include <stdlib.h>

int main()
{
	int n, m, i;
	int *a = NULL;
	scanf("%d %d",&n, &m);
	a = (int *)malloc(n * sizeof(int));
	m = m % n;    //当m >= n时
	for(i = m; i < n; i++)
	{
		scanf("%d", &a[i]);
	}
	for(i = 0; i < m; i++)
	{
		scanf("%d", &a[i]);
	}
	for(i = 0; i < n; i++)
	{
		if(i == n-1)
		{
			printf("%d\n", a[i]);
		}
		else
		{
			printf("%d ", a[i]);
		}
	}
	return 0;
}

```
