```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

int main()
{	
	int n, i, len, j, k;
	char str[40] = {'\0'};
	char s[4]; \\最大是三位数，位数+1，最后一位存'\0'。
	
	scanf("%d", &n);
	sprintf(s, "%d", n);
	len = strlen(s);
	j = (int)pow(10, len-1);
	while(n != 0)
	{
		i = n / j;
		switch(j)
		{
		case 100:
			for(k =0; k < i; k++)
			{
				printf("B");
			}
			break;
		case 10:
			for(k =0; k < i; k++)
			{
				printf("S");
			}
			break;
		case 1:
			for(k =0; k < i; k++)
			{
				printf("%d", k+1);
			}
			break;
		default:
			return 1;
		}
		n = n % j;
		j /= 10;
		
	}
	printf("\n");
	return 0;
}
```
